// apps/api/src/routes/contestRoutes.ts
// Contest management routes

import { Router } from 'express';
import { contestController } from '../controllers/contestController';
import { authenticate } from '../middlewares/authMiddleware';
import { asyncHandler } from '../middlewares/errorHandler';

const router = Router();

// =====================================
// Contest Routes
// =====================================

// @route   GET /api/v1/contests
// @desc    Get all upcoming contests
// @access  Public
router.get('/', asyncHandler(contestController.getAllContests));

// @route   GET /api/v1/contests/:id
// @desc    Get contest by ID
// @access  Public
router.get('/:id', asyncHandler(contestController.getContestById));

// @route   GET /api/v1/contests/platform/:platform
// @desc    Get contests by platform (leetcode/codeforces/codechef)
// @access  Public
router.get('/platform/:platform', asyncHandler(contestController.getContestsByPlatform));

// @route   POST /api/v1/contests/sync
// @desc    Trigger manual contest scraping
// @access  Private (Admin only in production)
router.post('/sync', authenticate, asyncHandler(contestController.triggerSync));

// @route   GET /api/v1/contests/upcoming
// @desc    Get upcoming contests (next 7 days)
// @access  Public
router.get('/filter/upcoming', asyncHandler(contestController.getUpcomingContests));

// @route   GET /api/v1/contests/today
// @desc    Get contests happening today
// @access  Public
router.get('/filter/today', asyncHandler(contestController.getTodayContests));

export default router;
