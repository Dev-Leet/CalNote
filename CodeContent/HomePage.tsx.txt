// apps/web/src/pages/HomePage.tsx
// Landing page

import { useEffect } from 'react';
import { useNavigate } from 'react-router-dom';
import { useAuthStore } from '../store/authStore';
import { Calendar, Bell, Code, Trophy } from 'lucide-react';

const HomePage = () => {
  const navigate = useNavigate();
  const { isAuthenticated } = useAuthStore();

  useEffect(() => {
    // Redirect to dashboard if already authenticated
    if (isAuthenticated) {
      navigate('/dashboard');
    }
  }, [isAuthenticated, navigate]);

  const handleGoogleLogin = () => {
    // Redirect to backend OAuth endpoint
    window.location.href = `${import.meta.env.VITE_API_URL}/api/v1/auth/google`;
  };

  return (
    <div className="min-h-screen bg-gradient-to-br from-indigo-50 via-white to-purple-50">
      {/* Header */}
      <header className="container mx-auto px-4 py-6">
        <div className="flex items-center justify-between">
          <div className="flex items-center gap-2">
            <Code className="w-8 h-8 text-indigo-600" />
            <h1 className="text-2xl font-bold text-gray-900">CP Calendar</h1>
          </div>
          <button
            onClick={handleGoogleLogin}
            className="px-6 py-2 bg-indigo-600 text-white rounded-lg hover:bg-indigo-700 transition"
          >
            Sign In
          </button>
        </div>
      </header>

      {/* Hero Section */}
      <main className="container mx-auto px-4 py-20">
        <div className="text-center max-w-4xl mx-auto">
          <h2 className="text-5xl font-bold text-gray-900 mb-6">
            Never Miss a Coding Contest
          </h2>
          <p className="text-xl text-gray-600 mb-12">
            Track upcoming contests from LeetCode, Codeforces, and CodeChef. Get
            smart reminders and sync directly to your Google Calendar.
          </p>

          <button
            onClick={handleGoogleLogin}
            className="px-8 py-4 bg-indigo-600 text-white text-lg rounded-xl hover:bg-indigo-700 transition shadow-lg hover:shadow-xl"
          >
            Get Started with Google
          </button>
        </div>

        {/* Features Grid */}
        <div className="grid md:grid-cols-2 lg:grid-cols-4 gap-8 mt-24">
          <FeatureCard
            icon={<Calendar className="w-12 h-12 text-indigo-600" />}
            title="All-in-One Calendar"
            description="View contests from multiple platforms in a unified calendar"
          />
          <FeatureCard
            icon={<Bell className="w-12 h-12 text-indigo-600" />}
            title="Smart Reminders"
            description="Get notified 15 min, 1 hour, and 1 day before contests"
          />
          <FeatureCard
            icon={<Trophy className="w-12 h-12 text-indigo-600" />}
            title="Multi-Platform"
            description="LeetCode, Codeforces, CodeChef — all in one place"
          />
          <FeatureCard
            icon={<Code className="w-12 h-12 text-indigo-600" />}
            title="Personal Notes"
            description="Take notes for each contest and track your progress"
          />
        </div>
      </main>

      {/* Footer */}
      <footer className="container mx-auto px-4 py-8 mt-20 border-t">
        <p className="text-center text-gray-600">
          © 2026 CP Calendar. Built for competitive programmers.
        </p>
      </footer>
    </div>
  );
};

interface FeatureCardProps {
  icon: React.ReactNode;
  title: string;
  description: string;
}

const FeatureCard = ({ icon, title, description }: FeatureCardProps) => (
  <div className="bg-white p-8 rounded-xl shadow-md hover:shadow-lg transition">
    <div className="mb-4">{icon}</div>
    <h3 className="text-xl font-semibold text-gray-900 mb-2">{title}</h3>
    <p className="text-gray-600">{description}</p>
  </div>
);

export default HomePage;
