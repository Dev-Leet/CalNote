// apps/api/src/scrapers/LeetCodeScraper.ts
// LeetCode contest scraper

import axios from 'axios';
import { Platform } from '@prisma/client';
import { logger } from '../config/logger';

export interface ScrapedContest {
  platform: Platform;
  externalId: string;
  name: string;
  url: string;
  startTime: Date;
  endTime: Date;
  duration: number;
  phase?: string;
  difficulty?: string;
}

export class LeetCodeScraper {
  private readonly API_URL = 'https://leetcode.com/graphql';

  /**
   * Scrape LeetCode contests
   */
  async scrapeContests(): Promise<ScrapedContest[]> {
    try {
      logger.info('Scraping LeetCode contests...');

      const query = `
        query upcomingContests {
          allContests {
            title
            titleSlug
            startTime
            duration
          }
        }
      `;

      const response = await axios.post(
        this.API_URL,
        { query },
        {
          headers: {
            'Content-Type': 'application/json',
            'User-Agent': 'CP Calendar Bot',
          },
        }
      );

      const rawContests = response.data?.data?.allContests || [];

      // Filter only upcoming contests
      const now = Date.now() / 1000;
      const upcomingContests = rawContests.filter(
        (c: any) => c.startTime > now
      );

      const contests: ScrapedContest[] = upcomingContests.map((contest: any) => {
        const startTime = new Date(contest.startTime * 1000);
        const duration = contest.duration; // in seconds
        const endTime = new Date(startTime.getTime() + duration * 1000);

        return {
          platform: Platform.LEETCODE,
          externalId: contest.titleSlug,
          name: contest.title,
          url: `https://leetcode.com/contest/${contest.titleSlug}`,
          startTime,
          endTime,
          duration: Math.floor(duration / 60), // Convert to minutes
        };
      });

      logger.info(`Found ${contests.length} LeetCode contests`);
      return contests;
    } catch (error) {
      logger.error('LeetCode scraper error:', error);
      return [];
    }
  }
}

export const leetCodeScraper = new LeetCodeScraper();
