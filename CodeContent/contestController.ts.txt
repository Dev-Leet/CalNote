// apps/api/src/controllers/contestController.ts
// Contest controller

import { Request, Response } from 'express';
import { contestService } from '../services/contestService';
import { scraperService } from '../services/scraperService';
import { logger } from '../config/logger';
import { BadRequestError } from '../middlewares/errorHandler';

class ContestController {
  /**
   * Get all contests with pagination and filters
   */
  async getAllContests(req: Request, res: Response): Promise<void> {
    try {
      const { page = '1', limit = '50', platform, upcoming } = req.query;

      const contests = await contestService.getContests({
        page: parseInt(page as string),
        limit: parseInt(limit as string),
        platform: platform as string,
        upcoming: upcoming === 'true',
      });

      res.status(200).json({
        success: true,
        data: contests,
      });
    } catch (error) {
      logger.error('Get all contests error:', error);
      res.status(500).json({
        success: false,
        message: 'Failed to fetch contests',
      });
    }
  }

  /**
   * Get contest by ID
   */
  async getContestById(req: Request, res: Response): Promise<void> {
    try {
      const { id } = req.params;
      const contest = await contestService.getContestById(id);

      if (!contest) {
        res.status(404).json({
          success: false,
          message: 'Contest not found',
        });
        return;
      }

      res.status(200).json({
        success: true,
        data: contest,
      });
    } catch (error) {
      logger.error('Get contest by ID error:', error);
      res.status(500).json({
        success: false,
        message: 'Failed to fetch contest',
      });
    }
  }

  /**
   * Get contests by platform
   */
  async getContestsByPlatform(req: Request, res: Response): Promise<void> {
    try {
      const { platform } = req.params;
      const validPlatforms = ['leetcode', 'codeforces', 'codechef'];

      if (!validPlatforms.includes(platform.toLowerCase())) {
        throw new BadRequestError('Invalid platform');
      }

      const contests = await contestService.getContestsByPlatform(
        platform.toUpperCase()
      );

      res.status(200).json({
        success: true,
        data: contests,
      });
    } catch (error) {
      logger.error('Get contests by platform error:', error);
      res.status(500).json({
        success: false,
        message: 'Failed to fetch contests',
      });
    }
  }

  /**
   * Get upcoming contests (next 7 days)
   */
  async getUpcomingContests(req: Request, res: Response): Promise<void> {
    try {
      const contests = await contestService.getUpcomingContests(7);

      res.status(200).json({
        success: true,
        data: contests,
      });
    } catch (error) {
      logger.error('Get upcoming contests error:', error);
      res.status(500).json({
        success: false,
        message: 'Failed to fetch upcoming contests',
      });
    }
  }

  /**
   * Get contests happening today
   */
  async getTodayContests(req: Request, res: Response): Promise<void> {
    try {
      const contests = await contestService.getTodayContests();

      res.status(200).json({
        success: true,
        data: contests,
      });
    } catch (error) {
      logger.error('Get today contests error:', error);
      res.status(500).json({
        success: false,
        message: 'Failed to fetch today contests',
      });
    }
  }

  /**
   * Trigger manual contest sync
   */
  async triggerSync(req: Request, res: Response): Promise<void> {
    try {
      // Trigger background scraping job
      await scraperService.scrapeAllPlatforms();

      res.status(200).json({
        success: true,
        message: 'Contest sync triggered successfully',
      });
    } catch (error) {
      logger.error('Trigger sync error:', error);
      res.status(500).json({
        success: false,
        message: 'Failed to trigger contest sync',
      });
    }
  }
}

export const contestController = new ContestController();
