// apps/api/src/services/contestService.ts
// Contest business logic service

import { prisma } from '../config/database';
import { Contest, Platform } from '@prisma/client';
import { CacheService } from '../config/redis';
import { logger } from '../config/logger';
import { startOfDay, endOfDay, addDays } from 'date-fns';

interface GetContestsOptions {
  page: number;
  limit: number;
  platform?: string;
  upcoming?: boolean;
}

class ContestService {
  private readonly CACHE_TTL = 300; // 5 minutes

  /**
   * Get contests with filters and pagination
   */
  async getContests(options: GetContestsOptions): Promise<{
    contests: Contest[];
    total: number;
    page: number;
    totalPages: number;
  }> {
    try {
      const { page, limit, platform, upcoming } = options;
      const skip = (page - 1) * limit;

      // Build where clause
      const where: any = {};

      if (platform) {
        where.platform = platform.toUpperCase() as Platform;
      }

      if (upcoming) {
        where.startTime = {
          gte: new Date(),
        };
      }

      // Check cache
      const cacheKey = `contests:${JSON.stringify({ page, limit, platform, upcoming })}`;
      const cached = await CacheService.get<any>(cacheKey);

      if (cached) {
        return cached;
      }

      // Fetch from database
      const [contests, total] = await Promise.all([
        prisma.contest.findMany({
          where,
          orderBy: { startTime: 'asc' },
          skip,
          take: limit,
        }),
        prisma.contest.count({ where }),
      ]);

      const result = {
        contests,
        total,
        page,
        totalPages: Math.ceil(total / limit),
      };

      // Cache result
      await CacheService.set(cacheKey, result, this.CACHE_TTL);

      return result;
    } catch (error) {
      logger.error('Get contests error:', error);
      throw error;
    }
  }

  /**
   * Get contest by ID
   */
  async getContestById(id: string): Promise<Contest | null> {
    try {
      const cacheKey = `contest:${id}`;
      const cached = await CacheService.get<Contest>(cacheKey);

      if (cached) {
        return cached;
      }

      const contest = await prisma.contest.findUnique({
        where: { id },
      });

      if (contest) {
        await CacheService.set(cacheKey, contest, this.CACHE_TTL);
      }

      return contest;
    } catch (error) {
      logger.error('Get contest by ID error:', error);
      throw error;
    }
  }

  /**
   * Get contests by platform
   */
  async getContestsByPlatform(platform: string): Promise<Contest[]> {
    try {
      const cacheKey = `contests:platform:${platform}`;
      const cached = await CacheService.get<Contest[]>(cacheKey);

      if (cached) {
        return cached;
      }

      const contests = await prisma.contest.findMany({
        where: {
          platform: platform as Platform,
          startTime: { gte: new Date() },
        },
        orderBy: { startTime: 'asc' },
      });

      await CacheService.set(cacheKey, contests, this.CACHE_TTL);

      return contests;
    } catch (error) {
      logger.error('Get contests by platform error:', error);
      throw error;
    }
  }

  /**
   * Get upcoming contests (next N days)
   */
  async getUpcomingContests(days: number = 7): Promise<Contest[]> {
    try {
      const cacheKey = `contests:upcoming:${days}`;
      const cached = await CacheService.get<Contest[]>(cacheKey);

      if (cached) {
        return cached;
      }

      const now = new Date();
      const endDate = addDays(now, days);

      const contests = await prisma.contest.findMany({
        where: {
          startTime: {
            gte: now,
            lte: endDate,
          },
        },
        orderBy: { startTime: 'asc' },
      });

      await CacheService.set(cacheKey, contests, this.CACHE_TTL);

      return contests;
    } catch (error) {
      logger.error('Get upcoming contests error:', error);
      throw error;
    }
  }

  /**
   * Get contests happening today
   */
  async getTodayContests(): Promise<Contest[]> {
    try {
      const cacheKey = 'contests:today';
      const cached = await CacheService.get<Contest[]>(cacheKey);

      if (cached) {
        return cached;
      }

      const now = new Date();
      const todayStart = startOfDay(now);
      const todayEnd = endOfDay(now);

      const contests = await prisma.contest.findMany({
        where: {
          OR: [
            {
              startTime: {
                gte: todayStart,
                lte: todayEnd,
              },
            },
            {
              AND: [
                { startTime: { lte: todayStart } },
                { endTime: { gte: todayEnd } },
              ],
            },
          ],
        },
        orderBy: { startTime: 'asc' },
      });

      await CacheService.set(cacheKey, contests, 60); // Cache for 1 minute

      return contests;
    } catch (error) {
      logger.error('Get today contests error:', error);
      throw error;
    }
  }

  /**
   * Create or update contest
   */
  async upsertContest(data: {
    platform: Platform;
    externalId: string;
    name: string;
    url: string;
    startTime: Date;
    endTime: Date;
    duration: number;
    phase?: string;
    difficulty?: string;
  }): Promise<Contest> {
    try {
      const contest = await prisma.contest.upsert({
        where: {
          platform_externalId: {
            platform: data.platform,
            externalId: data.externalId,
          },
        },
        update: {
          name: data.name,
          url: data.url,
          startTime: data.startTime,
          endTime: data.endTime,
          duration: data.duration,
          phase: data.phase,
          difficulty: data.difficulty,
          scrapedAt: new Date(),
        },
        create: data,
      });

      // Invalidate cache
      await CacheService.delPattern('contests:*');

      return contest;
    } catch (error) {
      logger.error('Upsert contest error:', error);
      throw error;
    }
  }
}

export const contestService = new ContestService();
