// apps/api/src/config/env.ts
// Environment configuration

import dotenv from 'dotenv';
import path from 'path';
import { z } from 'zod';

// Load environment variables
const envFile = process.env.NODE_ENV === 'test' ? '.env.test' : '.env.development';
dotenv.config({ path: path.resolve(__dirname, '../../', envFile) });

// Environment variable schema
const envSchema = z.object({
  // Node Environment
  NODE_ENV: z.enum(['development', 'production', 'test']).default('development'),
  PORT: z.string().transform(Number).default('4000'),

  // Database
  DATABASE_URL: z.string().url(),

  // Redis
  REDIS_URL: z.string().url(),

  // OAuth - Google
  GOOGLE_CLIENT_ID: z.string().min(1),
  GOOGLE_CLIENT_SECRET: z.string().min(1),
  GOOGLE_CALLBACK_URL: z.string().url(),

  // Google Calendar API
  GOOGLE_CALENDAR_API_KEY: z.string().optional(),

  // Ashna AI
  ASHNA_AI_API_KEY: z.string().min(1),
  ASHNA_AI_CONTEST_AGENT_URL: z.string().url().optional(),
  ASHNA_AI_NOTES_AGENT_URL: z.string().url().optional(),

  // JWT
  JWT_SECRET: z.string().min(32),
  JWT_EXPIRES_IN: z.string().default('7d'),

  // Session
  SESSION_SECRET: z.string().min(32),

  // Encryption
  ENCRYPTION_KEY: z.string().min(1),

  // Frontend URL
  FRONTEND_URL: z.string().url(),

  // Logging
  LOG_LEVEL: z.enum(['error', 'warn', 'info', 'debug']).default('info'),
});

// Validate environment variables
const parsed = envSchema.safeParse(process.env);

if (!parsed.success) {
  console.error('❌ Invalid environment variables:');
  console.error(parsed.error.flatten().fieldErrors);
  throw new Error('Invalid environment configuration');
}

export const config = {
  nodeEnv: parsed.data.NODE_ENV,
  port: parsed.data.PORT,
  
  // Database
  databaseUrl: parsed.data.DATABASE_URL,
  
  // Redis
  redisUrl: parsed.data.REDIS_URL,
  
  // Google OAuth
  google: {
    clientId: parsed.data.GOOGLE_CLIENT_ID,
    clientSecret: parsed.data.GOOGLE_CLIENT_SECRET,
    callbackUrl: parsed.data.GOOGLE_CALLBACK_URL,
    calendarApiKey: parsed.data.GOOGLE_CALENDAR_API_KEY,
  },
  
  // Ashna AI
  ashna: {
    apiKey: parsed.data.ASHNA_AI_API_KEY,
    contestAgentUrl: parsed.data.ASHNA_AI_CONTEST_AGENT_URL || 'https://api.ashna.ai/agents/contest-fetcher',
    notesAgentUrl: parsed.data.ASHNA_AI_NOTES_AGENT_URL || 'https://api.ashna.ai/agents/notes-generator',
  },
  
  // JWT
  jwt: {
    secret: parsed.data.JWT_SECRET,
    expiresIn: parsed.data.JWT_EXPIRES_IN,
  },
  
  // Session
  sessionSecret: parsed.data.SESSION_SECRET,
  
  // Encryption
  encryptionKey: parsed.data.ENCRYPTION_KEY,
  
  // Frontend
  frontendUrl: parsed.data.FRONTEND_URL,
  
  // Logging
  logLevel: parsed.data.LOG_LEVEL,
} as const;
