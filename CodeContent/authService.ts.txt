// apps/api/src/services/authService.ts
// Authentication service

import { prisma } from '../config/database';
import { User } from '@prisma/client';
import jwt from 'jsonwebtoken';
import { config } from '../config/env';
import CryptoJS from 'crypto-js';
import { UnauthorizedError } from '../middlewares/errorHandler';
import { logger } from '../config/logger';

interface GoogleUserData {
  googleId: string;
  email: string;
  name?: string;
  profilePicture?: string;
  accessToken: string;
  refreshToken: string;
}

class AuthService {
  /**
   * Encrypt sensitive data
   */
  private encrypt(text: string): string {
    return CryptoJS.AES.encrypt(text, config.encryptionKey).toString();
  }

  /**
   * Decrypt sensitive data
   */
  private decrypt(encryptedText: string): string {
    const bytes = CryptoJS.AES.decrypt(encryptedText, config.encryptionKey);
    return bytes.toString(CryptoJS.enc.Utf8);
  }

  /**
   * Find or create user from Google OAuth
   */
  async findOrCreateGoogleUser(userData: GoogleUserData): Promise<User> {
    try {
      // Check if user exists
      let user = await prisma.user.findUnique({
        where: { googleId: userData.googleId },
      });

      if (user) {
        // Update tokens
        user = await prisma.user.update({
          where: { id: user.id },
          data: {
            accessToken: this.encrypt(userData.accessToken),
            refreshToken: this.encrypt(userData.refreshToken),
            tokenExpiry: new Date(Date.now() + 3600 * 1000), // 1 hour
            lastLoginAt: new Date(),
          },
        });
      } else {
        // Create new user
        user = await prisma.user.create({
          data: {
            googleId: userData.googleId,
            email: userData.email,
            name: userData.name,
            profilePicture: userData.profilePicture,
            accessToken: this.encrypt(userData.accessToken),
            refreshToken: this.encrypt(userData.refreshToken),
            tokenExpiry: new Date(Date.now() + 3600 * 1000),
            lastLoginAt: new Date(),
          },
        });

        logger.info(`New user created: ${user.email}`);
      }

      return user;
    } catch (error) {
      logger.error('Find or create Google user error:', error);
      throw error;
    }
  }

  /**
   * Get user by ID
   */
  async getUserById(userId: string): Promise<Partial<User> | null> {
    try {
      const user = await prisma.user.findUnique({
        where: { id: userId },
        select: {
          id: true,
          email: true,
          name: true,
          profilePicture: true,
          timezone: true,
          notifyBefore15Min: true,
          notifyBefore1Hour: true,
          notifyBefore1Day: true,
          enabledPlatforms: true,
          createdAt: true,
          lastLoginAt: true,
        },
      });

      return user;
    } catch (error) {
      logger.error('Get user by ID error:', error);
      throw error;
    }
  }

  /**
   * Refresh access token
   */
  async refreshAccessToken(refreshToken: string): Promise<string> {
    try {
      // Verify refresh token
      const decoded = jwt.verify(refreshToken, config.jwt.secret) as any;

      // Find user
      const user = await prisma.user.findUnique({
        where: { id: decoded.userId },
      });

      if (!user) {
        throw new UnauthorizedError('User not found');
      }

      // Generate new access token
      const newToken = jwt.sign(
        { userId: user.id, email: user.email },
        config.jwt.secret,
        { expiresIn: config.jwt.expiresIn }
      );

      return newToken;
    } catch (error) {
      logger.error('Refresh access token error:', error);
      throw new UnauthorizedError('Invalid refresh token');
    }
  }

  /**
   * Get decrypted Google tokens for a user
   */
  async getGoogleTokens(userId: string): Promise<{
    accessToken: string;
    refreshToken: string;
  } | null> {
    try {
      const user = await prisma.user.findUnique({
        where: { id: userId },
        select: {
          accessToken: true,
          refreshToken: true,
        },
      });

      if (!user || !user.accessToken || !user.refreshToken) {
        return null;
      }

      return {
        accessToken: this.decrypt(user.accessToken),
        refreshToken: this.decrypt(user.refreshToken),
      };
    } catch (error) {
      logger.error('Get Google tokens error:', error);
      return null;
    }
  }
}

export const authService = new AuthService();
