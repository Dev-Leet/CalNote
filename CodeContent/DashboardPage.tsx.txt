// apps/web/src/pages/DashboardPage.tsx
// Main dashboard page

import { useQuery } from '@tanstack/react-query';
import { Calendar, Clock, Trophy } from 'lucide-react';
import { format, isPast, isFuture } from 'date-fns';
import axios from 'axios';
import ContestCard from '../components/dashboard/ContestCard';
import StatsCard from '../components/dashboard/StatsCard';

interface Contest {
  id: string;
  platform: string;
  name: string;
  url: string;
  startTime: string;
  endTime: string;
  duration: number;
  phase?: string;
  difficulty?: string;
}

const DashboardPage = () => {
  // Fetch upcoming contests
  const { data: upcomingData, isLoading } = useQuery({
    queryKey: ['contests', 'upcoming'],
    queryFn: async () => {
      const response = await axios.get('/api/v1/contests/filter/upcoming');
      return response.data.data as Contest[];
    },
  });

  // Fetch today's contests
  const { data: todayData } = useQuery({
    queryKey: ['contests', 'today'],
    queryFn: async () => {
      const response = await axios.get('/api/v1/contests/filter/today');
      return response.data.data as Contest[];
    },
  });

  const upcomingContests = upcomingData || [];
  const todayContests = todayData || [];
  const liveContests = todayContests.filter((c) => {
    const now = new Date();
    return new Date(c.startTime) <= now && new Date(c.endTime) >= now;
  });

  if (isLoading) {
    return (
      <div className="flex items-center justify-center h-96">
        <div className="animate-spin rounded-full h-12 w-12 border-b-2 border-indigo-600"></div>
      </div>
    );
  }

  return (
    <div className="space-y-8">
      {/* Header */}
      <div>
        <h1 className="text-3xl font-bold text-gray-900">Dashboard</h1>
        <p className="text-gray-600 mt-2">
          Track upcoming contests and stay prepared
        </p>
      </div>

      {/* Stats Cards */}
      <div className="grid md:grid-cols-3 gap-6">
        <StatsCard
          icon={<Trophy className="w-6 h-6 text-indigo-600" />}
          title="Live Contests"
          value={liveContests.length}
          subtitle="Happening now"
          color="indigo"
        />
        <StatsCard
          icon={<Calendar className="w-6 h-6 text-green-600" />}
          title="Today's Contests"
          value={todayContests.length}
          subtitle="Scheduled today"
          color="green"
        />
        <StatsCard
          icon={<Clock className="w-6 h-6 text-purple-600" />}
          title="Upcoming (7 days)"
          value={upcomingContests.length}
          subtitle="In the next week"
          color="purple"
        />
      </div>

      {/* Live Contests */}
      {liveContests.length > 0 && (
        <section>
          <h2 className="text-2xl font-semibold text-gray-900 mb-4">
            🔴 Live Contests
          </h2>
          <div className="grid gap-4">
            {liveContests.map((contest) => (
              <ContestCard key={contest.id} contest={contest} isLive />
            ))}
          </div>
        </section>
      )}

      {/* Upcoming Contests */}
      <section>
        <h2 className="text-2xl font-semibold text-gray-900 mb-4">
          Upcoming Contests
        </h2>
        {upcomingContests.length === 0 ? (
          <div className="text-center py-12 bg-gray-50 rounded-lg">
            <Calendar className="w-16 h-16 text-gray-400 mx-auto mb-4" />
            <p className="text-gray-600">No upcoming contests in the next 7 days</p>
          </div>
        ) : (
          <div className="grid gap-4">
            {upcomingContests.slice(0, 10).map((contest) => (
              <ContestCard key={contest.id} contest={contest} />
            ))}
          </div>
        )}
      </section>
    </div>
  );
};

export default DashboardPage;
