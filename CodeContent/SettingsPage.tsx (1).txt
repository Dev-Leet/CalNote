// frontend/src/pages/SettingsPage.tsx
import React, { useState, useEffect } from 'react';
import { useMutation, useQuery, useQueryClient } from '@tanstack/react-query';
import { Save, Bell, Globe, Link as LinkIcon } from 'lucide-react';
import { getSettings, updateSettings, syncGoogleCalendar } from '../api/settings';
import { toast } from 'react-hot-toast';

interface Settings {
  notificationsEnabled: boolean;
  notifyBefore15Min: boolean;
  notifyBefore1Hour: boolean;
  notifyBefore1Day: boolean;
  enabledPlatforms: string[];
  timezone: string;
  googleCalendarConnected: boolean;
}

const SettingsPage: React.FC = () => {
  const queryClient = useQueryClient();
  const [formData, setFormData] = useState<Settings>({
    notificationsEnabled: true,
    notifyBefore15Min: true,
    notifyBefore1Hour: true,
    notifyBefore1Day: false,
    enabledPlatforms: ['leetcode', 'codeforces', 'codechef'],
    timezone: Intl.DateTimeFormat().resolvedOptions().timeZone,
    googleCalendarConnected: false,
  });

  const { data: settings, isLoading } = useQuery({
    queryKey: ['settings'],
    queryFn: getSettings,
  });

  useEffect(() => {
    if (settings) {
      setFormData(settings);
    }
  }, [settings]);

  const updateMutation = useMutation({
    mutationFn: updateSettings,
    onSuccess: () => {
      queryClient.invalidateQueries({ queryKey: ['settings'] });
      toast.success('Settings saved successfully');
    },
    onError: () => {
      toast.error('Failed to save settings');
    },
  });

  const syncCalendarMutation = useMutation({
    mutationFn: syncGoogleCalendar,
    onSuccess: () => {
      toast.success('Google Calendar synced successfully');
    },
    onError: () => {
      toast.error('Failed to sync Google Calendar');
    },
  });

  const handleSubmit = (e: React.FormEvent) => {
    e.preventDefault();
    updateMutation.mutate(formData);
  };

  const togglePlatform = (platform: string) => {
    setFormData((prev) => ({
      ...prev,
      enabledPlatforms: prev.enabledPlatforms.includes(platform)
        ? prev.enabledPlatforms.filter((p) => p !== platform)
        : [...prev.enabledPlatforms, platform],
    }));
  };

  const handleConnectGoogleCalendar = () => {
    // Redirect to Google OAuth
    window.location.href = `${process.env.REACT_APP_API_URL}/auth/google/calendar`;
  };

  if (isLoading) {
    return (
      <div className="flex items-center justify-center h-96">
        <div className="animate-spin rounded-full h-12 w-12 border-b-2 border-indigo-600" />
      </div>
    );
  }

  return (
    <div className="space-y-6">
      {/* Header */}
      <div>
        <h1 className="text-3xl font-bold text-gray-900">Settings</h1>
        <p className="text-gray-600 mt-1">
          Manage your preferences and integrations
        </p>
      </div>

      <form onSubmit={handleSubmit} className="space-y-6">
        {/* Notifications Section */}
        <div className="bg-white rounded-lg shadow-sm border border-gray-200 p-6">
          <div className="flex items-center space-x-3 mb-6">
            <Bell className="h-6 w-6 text-indigo-600" />
            <h2 className="text-xl font-semibold text-gray-900">
              Notification Preferences
            </h2>
          </div>

          <div className="space-y-4">
            <label className="flex items-center justify-between cursor-pointer">
              <div>
                <p className="font-medium text-gray-900">Enable Notifications</p>
                <p className="text-sm text-gray-600">
                  Receive contest reminders via email
                </p>
              </div>
              <input
                type="checkbox"
                checked={formData.notificationsEnabled}
                onChange={(e) =>
                  setFormData({ ...formData, notificationsEnabled: e.target.checked })
                }
                className="h-5 w-5 text-indigo-600 focus:ring-indigo-500 border-gray-300 rounded"
              />
            </label>

            {formData.notificationsEnabled && (
              <div className="pl-6 space-y-3 border-l-2 border-gray-200">
                <label className="flex items-center justify-between cursor-pointer">
                  <span className="text-gray-700">15 minutes before</span>
                  <input
                    type="checkbox"
                    checked={formData.notifyBefore15Min}
                    onChange={(e) =>
                      setFormData({ ...formData, notifyBefore15Min: e.target.checked })
                    }
                    className="h-4 w-4 text-indigo-600 focus:ring-indigo-500 border-gray-300 rounded"
                  />
                </label>
                <label className="flex items-center justify-between cursor-pointer">
                  <span className="text-gray-700">1 hour before</span>
                  <input
                    type="checkbox"
                    checked={formData.notifyBefore1Hour}
                    onChange={(e) =>
                      setFormData({ ...formData, notifyBefore1Hour: e.target.checked })
                    }
                    className="h-4 w-4 text-indigo-600 focus:ring-indigo-500 border-gray-300 rounded"
                  />
                </label>
                <label className="flex items-center justify-between cursor-pointer">
                  <span className="text-gray-700">1 day before</span>
                  <input
                    type="checkbox"
                    checked={formData.notifyBefore1Day}
                    onChange={(e) =>
                      setFormData({ ...formData, notifyBefore1Day: e.target.checked })
                    }
                    className="h-4 w-4 text-indigo-600 focus:ring-indigo-500 border-gray-300 rounded"
                  />
                </label>
              </div>
            )}
          </div>
        </div>

        {/* Platforms Section */}
        <div className="bg-white rounded-lg shadow-sm border border-gray-200 p-6">
          <h2 className="text-xl font-semibold text-gray-900 mb-6">
            Enabled Platforms
          </h2>

          <div className="grid grid-cols-1 md:grid-cols-3 gap-4">
            {[
              { id: 'leetcode', name: 'LeetCode', color: 'yellow' },
              { id: 'codeforces', name: 'Codeforces', color: 'blue' },
              { id: 'codechef', name: 'CodeChef', color: 'orange' },
            ].map((platform) => (
              <label
                key={platform.id}
                className="flex items-center space-x-3 p-4 border border-gray-200 rounded-lg cursor-pointer hover:bg-gray-50 transition"
              >
                <input
                  type="checkbox"
                  checked={formData.enabledPlatforms.includes(platform.id)}
                  onChange={() => togglePlatform(platform.id)}
                  className="h-5 w-5 text-indigo-600 focus:ring-indigo-500 border-gray-300 rounded"
                />
                <span className="font-medium text-gray-900">{platform.name}</span>
              </label>
            ))}
          </div>
        </div>

        {/* Timezone Section */}
        <div className="bg-white rounded-lg shadow-sm border border-gray-200 p-6">
          <div className="flex items-center space-x-3 mb-6">
            <Globe className="h-6 w-6 text-indigo-600" />
            <h2 className="text-xl font-semibold text-gray-900">Timezone</h2>
          </div>

          <select
            value={formData.timezone}
            onChange={(e) => setFormData({ ...formData, timezone: e.target.value })}
            className="w-full px-4 py-3 border border-gray-300 rounded-lg focus:ring-2 focus:ring-indigo-500 focus:border-transparent"
          >
            <option value="UTC">UTC (Coordinated Universal Time)</option>
            <option value="America/New_York">America/New York (EST/EDT)</option>
            <option value="America/Chicago">America/Chicago (CST/CDT)</option>
            <option value="America/Denver">America/Denver (MST/MDT)</option>
            <option value="America/Los_Angeles">America/Los Angeles (PST/PDT)</option>
            <option value="Europe/London">Europe/London (GMT/BST)</option>
            <option value="Europe/Paris">Europe/Paris (CET/CEST)</option>
            <option value="Asia/Kolkata">Asia/Kolkata (IST)</option>
            <option value="Asia/Shanghai">Asia/Shanghai (CST)</option>
            <option value="Asia/Tokyo">Asia/Tokyo (JST)</option>
            <option value="Australia/Sydney">Australia/Sydney (AEST/AEDT)</option>
          </select>
        </div>

        {/* Google Calendar Integration */}
        <div className="bg-white rounded-lg shadow-sm border border-gray-200 p-6">
          <div className="flex items-center space-x-3 mb-6">
            <LinkIcon className="h-6 w-6 text-indigo-600" />
            <h2 className="text-xl font-semibold text-gray-900">
              Google Calendar Integration
            </h2>
          </div>

          {formData.googleCalendarConnected ? (
            <div className="space-y-3">
              <p className="text-green-600 font-medium">
                ✓ Connected to Google Calendar
              </p>
              <button
                type="button"
                onClick={() => syncCalendarMutation.mutate()}
                disabled={syncCalendarMutation.isPending}
                className="px-4 py-2 bg-indigo-600 text-white rounded-lg hover:bg-indigo-700 disabled:opacity-50 transition"
              >
                {syncCalendarMutation.isPending ? 'Syncing...' : 'Sync Now'}
              </button>
            </div>
          ) : (
            <div className="space-y-3">
              <p className="text-gray-600">
                Connect your Google Calendar to automatically sync contest schedules
              </p>
              <button
                type="button"
                onClick={handleConnectGoogleCalendar}
                className="px-4 py-2 bg-indigo-600 text-white rounded-lg hover:bg-indigo-700 transition"
              >
                Connect Google Calendar
              </button>
            </div>
          )}
        </div>

        {/* Save Button */}
        <div className="flex justify-end">
          <button
            type="submit"
            disabled={updateMutation.isPending}
            className="flex items-center space-x-2 px-6 py-3 bg-indigo-600 text-white rounded-lg hover:bg-indigo-700 disabled:opacity-50 font-medium transition"
          >
            <Save className="h-5 w-5" />
            <span>{updateMutation.isPending ? 'Saving...' : 'Save Settings'}</span>
          </button>
        </div>
      </form>
    </div>
  );
};

export default SettingsPage;
