// backend/src/cron/scrapeContests.ts
import cron from 'node-cron';
import axios from 'axios';
import { Contest } from '../models/Contest';
import { logger } from '../utils/logger';

interface ContestData {
  name: string;
  url: string;
  start_time: string;
  end_time: string;
  duration: string;
  site: string;
  in_24_hours: string;
  status: string;
}

/**
 * Scrape contests from Kontests.net API
 * Supports LeetCode, Codeforces, CodeChef
 */
export async function scrapeContests(): Promise<void> {
  try {
    logger.info('Starting contest scraping...');

    const platforms = ['leetcode', 'codeforces', 'codechef'];
    const allContests: any[] = [];

    for (const platform of platforms) {
      try {
        const response = await axios.get<ContestData[]>(
          `https://kontests.net/api/v1/${platform}`
        );

        const contests = response.data.map((contest) => ({
          name: contest.name,
          platform: platform,
          url: contest.url,
          startTime: new Date(contest.start_time),
          endTime: new Date(contest.end_time),
          duration: contest.duration,
          status: contest.status,
        }));

        allContests.push(...contests);
        logger.info(`Scraped ${contests.length} contests from ${platform}`);
      } catch (error) {
        logger.error(`Error scraping ${platform}:`, error);
      }
    }

    // Upsert contests into database
    for (const contest of allContests) {
      await Contest.findOneAndUpdate(
        { url: contest.url },
        contest,
        { upsert: true, new: true }
      );
    }

    logger.info(`Total contests updated: ${allContests.length}`);
  } catch (error) {
    logger.error('Error in scrapeContests:', error);
    throw error;
  }
}

/**
 * Schedule contest scraping every 6 hours
 */
export function scheduleContestScraping(): void {
  // Run every 6 hours: at 00:00, 06:00, 12:00, 18:00
  cron.schedule('0 */6 * * *', async () => {
    logger.info('Running scheduled contest scraping...');
    await scrapeContests();
  });

  // Run immediately on startup
  scrapeContests();

  logger.info('Contest scraping scheduled: every 6 hours');
}

/**
 * Alternative: Scrape from Clist.by API (requires API key)
 */
export async function scrapeFromClist(apiKey: string): Promise<void> {
  try {
    const response = await axios.get('https://clist.by/api/v2/contest/', {
      params: {
        resource: 'leetcode.com,codeforces.com,codechef.com',
        start__gt: new Date().toISOString(),
        limit: 100,
        order_by: 'start',
      },
      headers: {
        Authorization: `ApiKey ${apiKey}`,
      },
    });

    const contests = response.data.objects;
    logger.info(`Scraped ${contests.length} contests from Clist.by`);

    // Process and save contests
    for (const contest of contests) {
      await Contest.findOneAndUpdate(
        { url: contest.href },
        {
          name: contest.event,
          platform: contest.resource,
          url: contest.href,
          startTime: new Date(contest.start),
          endTime: new Date(contest.end),
          duration: contest.duration,
          status: 'upcoming',
        },
        { upsert: true, new: true }
      );
    }
  } catch (error) {
    logger.error('Error scraping from Clist.by:', error);
    throw error;
  }
}
